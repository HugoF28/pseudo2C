VARIABLE name1 is string
VARIABLE name2 is string
VARIABLE var1 is int
VARIABLE var2 is int
VARIABLE var1ans is int
VARIABLE var2ans is int
VARIABLE turn is int
VARIABLE i is int

SET turn to 0
SET var1ans to -1
SET var2ans to -1

PRINT "=== Guessing game ==="

PRINT "Enter player 1 name: "
GET name1
PRINT "Player 1 enter number(1-10):"
GET var1

FOR i = 0 UNTIL i < 30
	PRINT "\n"
ENDFOR

PRINT "Enter player 2 name: "
GET name2

PRINT "Player 2 enter number(1-10):"
GET var2

FOR i = 0 UNTIL i < 30
	PRINT "\n"
ENDFOR

PRINT "It's time to guess!"

WHILE var1ans != var1 && var2ans != var2
	IF turn == 1
		PRINT name1
		GET var2ans
		
		IF var2ans != var2
			PRINT "wrong answer!"
		ENDIF
	ENDIF
	ELSE
		PRINT name2
		GET var1ans
		
		IF var1ans != var1
			PRINT "wrong answer!"
		ENDIF

	ENDELSE
	
	IF turn == 0
		SET turn TO 1
	ENDIF	
	ELSE
		SET turn TO 0
	ENDELSE
ENDWHILE

PRINT "You are the winner"
